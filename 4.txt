clc;
clear all;
close all;

f1 = 60;
f2 = 120;
Fs = 480;
N  = 64;

t = (0:N-1) / Fs;

x = sin(2*pi*f1*t) + sin(2*pi*f2*t);

X = fft(x, N);

f = (0:N/2-1) * (Fs/N);

amplitude = (2/N) * abs(X(1:N/2));

figure;
stem(f, amplitude, 'filled');
xlabel('Frequency (Hz)');
ylabel('Amplitude');
title('Single-Sided FFT Spectrum');
grid on;

disp('Frequency bins (Hz):');
disp(f);
disp('Amplitude values:');
disp(amplitude);
