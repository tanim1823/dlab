clc;
clear all;
close all;

% Input signals
x = [1, 1, 2, 2];
h = [1, 2, 3, 4];

% Convolution
y = conv(x, h);

% Plot
figure;

subplot(3,1,1);
stem(x);
ylabel('Amplitude -->');
xlabel('(a) n -->');
title('Signal x[n]');

subplot(3,1,2);
stem(h);
ylabel('Amplitude -->');
xlabel('(b) n -->');
title('Signal h[n]');

subplot(3,1,3);
stem(y);
ylabel('Amplitude -->');
xlabel('(c) n -->');
title('Convolved Output y[n]');

disp('The resultant signal is:');
disp(y);
