clc;
clear all;
close all;

pkg load signal;

x = [1, 0, 0, 1];
h = [4, 3, 2, 1];

y = xcorr(x, h);

figure;

subplot(3, 1, 1);
stem(x);
ylabel('Amplitude -->');
xlabel('(a) n -->');
title('1st Sequence x(n)');

subplot(3, 1, 2);
stem(h);
ylabel('Amplitude -->');
xlabel('(b) n -->');
title('2nd Sequence h(n)');

subplot(3, 1, 3);
stem(y);
ylabel('Amplitude -->');
xlabel('(c) n -->');
title('Cross-correlation Output y(n)');

disp('The cross-correlation output is:');
disp(y);
